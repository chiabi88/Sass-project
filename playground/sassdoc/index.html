  px: 1px, pt: 1pt, pc: 1pc, in: 1in, mm: 1mm, cm: 1cm, // absolute length
  em: 1em, rem: 1rem, ch: 1ch, ex: 1ex,                 // relative length - font based

  hz: 1Hz, khz: 1kHz,                                   // frequency
