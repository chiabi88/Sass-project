<ul class="guide-burger">
	<li>
		<button class="burger--cross"><span>버튼1:크로스</span></button>
	</li>
	<li>
		<button class="burger--minus"><span>버튼2:마이너스</span></button>

	<li>
		<button class="burger--arrow"><span>버튼3:화살표</span></button>
	</li>

scss

.burger {
	@include burger-menu (
		$burger-width: 50px,
		$burger-height: 50px,
		$burger-side: 15px,
		$burger-gutter: 4px,
		$burger-thickness: 2px,
		$burger-color: lighten($color-primary, 50%),
		$burger-transition-duration: 0.3s,
		$burger-border-radius: 0
	);
	&--cross:hover {
		@include burger-menu--cross ( rgb(240,240,120) );
	}
	&--minus:hover {
		@include burger-menu--minus ( rgb(120,240,240) );
	}
	&--arrow:hover {
		@include burger-menu--arrow ( rgb(240,120,240) );
	}
}

# Check Box

html

<div class="guide-checkbox">
	<div class="checkbox">
		<input type="checkbox" id="check1">
		<label for="check1"><span></span>체크 박스 라벨1</label>
	</div>
	<div class="checkbox">
		<input type="checkbox" id="check2" checked>
		<label for="check2"><span></span>체크 박스  라벨2</label>
	</div>
	<div class="checkbox">
		<input type="checkbox" id="check3" disabled>
		<label for="check3"><span></span>체크 박스 라벨3</label>
	</div>
	<div class="checkbox">
		<input type="checkbox" id="check3" checked disabled>
		<label for="check3"><span></span>체크 박스 라벨3</label>
	</div>
</div>

scss

$guide-checkbox: (
	'checkbox': (
		'margin': 10px,
		'size': 30px,
		'border-width': 1px,
		'border-color': lighten($color-primary, 20%),
		'background': lighten($color-primary, 45%)
	),
	'mark': (
		'size': 10px,
		'border-width': 1px,
		'color': rgb(255,255,255)
	),
	'checked': (
		'border-width': null,
		'border-color': darken($color-primary, 5%),
		'background': darken($color-primary, 10%)
	),  
	'disabled': (
		'font-color': rgb(153,153,153),
		'mark-color': rgb(255,255,255),
		'border-color': rgb(238,238,238),
		'background': rgb(153,153,153)
	)
);

.checkbox {
	font-size: 14px;
	@include checkbox ($guide-checkbox);
}
